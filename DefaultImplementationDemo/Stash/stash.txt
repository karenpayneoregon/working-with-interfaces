namespace DefaultImplementationDemo;

internal partial class Program
{
    static void Main(string[] args)
    {

        Demo d = new();
        d.SomeMethod();
        //d.WillNotBreakExistingApplications();


        // Create an instance of Demo
        ISample demoInstance = new Demo();
        // Call the default implementation explicitly
        demoInstance.WillNotBreakExistingApplications(); // Calls Demo class implementation
        // To call the default implementation, use an explicit cast to the interface
        demoInstance.WillNotBreakExistingApplications();

        Console.ReadLine();
    }
}

internal interface ISample
{
    void SomeMethod(); //method having only declaration
    public void WillNotBreakExistingApplications()
    {
        Console.WriteLine("Here in the interface");
    }
}

public class Demo : ISample
{
    public void SomeMethod()
    {
        Console.WriteLine("Some method");
    }
